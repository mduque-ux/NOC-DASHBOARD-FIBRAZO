// ══════════════════════════════════════ // CONEXIÓN REAL-TIME (DASHBOARD SIDE) // ══════════════════════════════════════ async function refreshRealData() { // ⚠️ SUSTITUYE ESTA URL POR TU URL DE "DEPLOY" REAL const WEB_APP_URL = "https://script.google.com/macros/s/AKfycb...TU_ID.../exec"; try { console.log("Iniciando sincronización..."); const response = await fetch(WEB_APP_URL); const data = await response.json(); if (data.status === "success") { // Actualizar Contadores document.getElementById('k-odoo').textContent = data.odoo.totalN; document.getElementById('k-tv').textContent = data.tv.canalesDown.length; document.getElementById('k-mops').textContent = data.mops.lista.length; document.getElementById('k-noc').textContent = data.asignados.tickets.length; // Actualizar Lista de TV const tvList = document.getElementById('tv-list'); if(tvList) { tvList.innerHTML = data.tv.canalesDown.map(c => `
● ${c.nombre} (Uptime: ${c.uptime}%)
`).join('') || "✅ Todos los canales UP"; } console.log("Última actualización:", data.actualizado); } } catch (e) { console.error("Error conectando con el Script:", e); } } // Cargar al iniciar window.onload = refreshRealData;
