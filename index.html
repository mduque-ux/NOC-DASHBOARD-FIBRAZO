// --- CONFIGURACIÓN Y ESTADO --- const CONFIG = { ingenierosSinNoche: ['Joseph Montenegro'], slaCriticoHoras: 24, autoSaveKey: 'noc_dashboard_draft' }; // --- RELOJ Y FECHAS --- function updateClock() { const now = new Date(); const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' }; document.getElementById('hdrDate').textContent = now.toLocaleDateString('es-CO', options); document.getElementById('liveClock').textContent = now.toLocaleTimeString('es-CO') + ' COT'; } // --- LÓGICA DE FORMULARIO --- function onIngChange() { const ing = document.getElementById('sel-ing').value; const selTurno = document.getElementById('sel-turno'); const warn = document.getElementById('turno-warn'); // Filtrar turnos permitidos const turnosDiurnos = ['06:00 - 12:00', '12:00 - 18:00']; const turnosNocturnos = ['18:00 - 00:00', '00:00 - 06:00']; selTurno.innerHTML = ''; const lista = CONFIG.ingenierosSinNoche.includes(ing) ? turnosDiurnos : [...turnosDiurnos, ...turnosNocturnos]; lista.forEach(t => { const opt = document.createElement('option'); opt.value = opt.textContent = t; selTurno.appendChild(opt); }); warn.style.display = CONFIG.ingenierosSinNoche.includes(ing) ? 'block' : 'none'; document.getElementById('ing-disp').textContent = ing || 'Selecciona ingeniero'; saveDraft(); } // --- AUTO-GUARDADO (LOCAL STORAGE) --- function saveDraft() { const draft = { ing: document.getElementById('sel-ing').value, desc: document.getElementById('desc-ent').value, obs: document.getElementById('obs-msg').value, fecha: document.getElementById('sel-fecha').value }; localStorage.setItem(CONFIG.autoSaveKey, JSON.stringify(draft)); } function loadDraft() { const saved = localStorage.getItem(CONFIG.autoSaveKey); if (saved) { const data = JSON.parse(saved); if (data.ing) { document.getElementById('sel-ing').value = data.ing; onIngChange(); } document.getElementById('desc-ent').value = data.desc || ''; document.getElementById('obs-msg').value = data.obs || ''; if (data.fecha) document.getElementById('sel-fecha').value = data.fecha; } } // --- UTILIDADES --- function copyToClipboard() { const ing = document.getElementById('sel-ing').value; const desc = document.getElementById('desc-ent').value; const texto = `*REPORTE NOC FIBRAZO*\n👤 *Ing:* ${ing}\n📝 *Resumen:* ${desc}`; navigator.clipboard.writeText(texto).then(() => { toast('Copiado para WhatsApp/Slack ✓', 'ok'); }); } function toast(msg, tipo) { const t = document.getElementById('toast'); t.style.background = tipo === 'ok' ? '#1DB954' : '#EF4444'; t.textContent = msg; t.style.display = 'block'; setTimeout(() => t.style.display = 'none', 3000); } // --- INICIALIZACIÓN --- document.addEventListener('DOMContentLoaded', () => { updateClock(); setInterval(updateClock, 1000); // Cargar datos previos loadDraft(); // Event Listeners document.getElementById('sel-ing').addEventListener('change', onIngChange); document.getElementById('desc-ent').addEventListener('input', saveDraft); document.getElementById('obs-msg').addEventListener('input', saveDraft); // Botón de guardar (ahora limpia el draft al terminar) document.getElementById('btn-guardar').addEventListener('click', () => { if (!document.getElementById('sel-ing').value) { return toast('Selecciona un ingeniero primero', 'error'); } toast('Registro guardado y sincronizado', 'ok'); localStorage.removeItem(CONFIG.autoSaveKey); }); // Efecto de refresco visual document.getElementById('btn-refresh').addEventListener('click', function() { this.textContent = 'Sincronizando...'; setTimeout(() => { this.textContent = '↻ Actualizar ahora'; toast('Datos de red actualizados', 'ok'); }, 800); }); });
